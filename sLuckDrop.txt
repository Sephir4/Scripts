function	script	sLuckDrop	{
	for(set @i,0;@i < getargcount();set @i,@i+3){
		set @rd, rand(1000);
		if(getarg(@i+2) >= @rd){
			getitem getarg(@i),getarg(@i+1);
			return;
		}
	}
	return;
}
